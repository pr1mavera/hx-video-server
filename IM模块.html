# 登录

# 腾讯IM登录

通过包装原来的腾讯IM登录接口，添加默认消息接收器 onMsgNotify（可通过参数重写），将来自 webSocket 的远程消息（c2c消息、系统消息）转移至统一的消息事件总线，分离登录逻辑与后续的消息处理逻辑，/src/module/im/login.js

// 总线实例索引
let msgNotifySubjectSub = null;
// 消息订阅并解析，推送至各消息管道
const msgNotifySubjectInit = () => {
    // 若初始化过则直接返回之前的subject
    if (msgNotifySubjectSub) return msgNotifySubjectSub;

    // 初始化subject
    msgNotifySubjectSub = new Subject().pipe(
        flatMap(msg => parseMsgs(msg)) // 此处解析进入的消息
    );
    // 在内部订阅
    msgNotifySubjectSub.subscribe(m => {
        // 消息 type$ 类型分流：c2c消息、系统消息、指令消息
        getMsgStream$(m.type$).next(m);
    });

    return msgNotifySubjectSub;
};
/**
 * 登录 | public
 */
const login = ({
    listeners = {},
    options = {
        'isAccessFormalEnv': true, // 是否访问正式环境，默认访问正式，选填
        'isLogOn': false // 是否开启控制台打印日志，默认开启，选填
    }
} = {}) => {

    /**
     * 默认的IM消息解析（可在外部自定义消息解析方式）
     * 只在login模块内可用
     */
    const _onMsgNotify = msgs => {
        // 此处直接将拿到的消息推送至总线
        msgNotifySubjectInit().next(msgs);
    };

    const accountInfo = {
        // 用户登录相关参数配置
    };
    // 调用封装好的登录接口
    return IMlogin(accountInfo, { onMsgNotify: _onMsgNotify, ...listeners }, options);
};

# 消息解析

消息解析的部分逻辑在 /src/module/im/parser.js 中

# 解析流程

  1. 根据 fromAccount 字段获取解析器
  2. 分别解析c2c消息和系统消息，并包装对应的 type$ 标明消息类型
  3. 根据解析后的消息体分离出指令消息

# 消息分类

  • 根据消息体上的 fromAccount 字段判断是否是系统消息
    • 是系统消息 -> system
    • 不是系统消息，先暂定消息为c2c消息，将消息正常解析，判断解析后消息上的 msgType 属性判断是否在指令集 directiveTypes 中
      • 在指令集中 -> directive
      • 不在指令集中 -> C2C
  • 解析后的消息会带上 type$ 标明消息类型
// 指令消息类型
const directiveTypes = [
    // ...
];

const _msgStreamTypes = {
    'C2C': new Subject(),
    'system': new Subject(),
    'directive': new Subject()
};

# 消息发送

消息发送分为「系统消息」和「c2c消息」

# 系统消息

即简单的包装消息体，通过接口发送至node层，node调用腾讯IM系统消息推送接口，将接口调用返回的 promise 直接return出去

# c2c消息

c2c消息的发送则稍微复杂一些，根据消息类型的不同分离出了不同的API，同时将聊天相关消息发送的回调处理分离，可以方便的在统一的回调函数中，做消息发送失败的业务处理

/**
 * C2C消息总线
 */
const _msg$ = new Subject();
// 聊天相关消息的发送会将消息推送至 _msg$ 流中，而不是直接的调发送接口
/**
 * C2C消息发送状态通知流，用于处理C2C消息发的成功与失败
 */
const sendMsgsFeedback$ = combineLatest(_msg$, cs$.getCS_$()).pipe(
    // 过滤掉未注册坐席的异常操作 !!! Side effect !!!
    filter(([ , cs ]) => _isCsRegister(cs)),
    // 过滤掉由坐席状态变更导致的总线通知（拿到的消息的流与上一条时间戳一致，说明不是消息的流更新）
    distinctUntilChanged(([ p_msg, ], [ n_msg, ]) => p_msg.timestamp === n_msg.timestamp),
    // 格式化C2C消息体
    map(data => _initMsg(data)),
    // webim发送消息
    flatMap(msgBody => from(IMsendNormalMsg(msgBody)).pipe(catchError(err => of(err))))
);

消息发送之前会根据不同消息类型做不同的参数规则校验，并且这个参数是灵活可配的

// ./C2C/index.js
/**
 * C2C消息生成器，方法只将消息体包装成流推送至消息总线，统一发送
 */
const { /* 一系列消息类型 */ } =
    [ /* 不同消息类型附带的数据 */ ].reduce(
        (group, [ /* 其他数据 */, formatArgs ]) => ({
            [fn_name]: args => {
                let msg;
                msg = {
                    /* 其他数据 */,
                    ...formatArgs(args) // 在此处对API调用方传入的参数做校验处理
                };
                // 推送至消息总线
                _msg$.next(msg);
            },
            ...group
        }), {}
    );

// ./C2C/methods.js
// 定义各个消息发送的默认参数及其传入数据的格式化过程
export const sendTextFormatter = {
    option: [ msgStatus.msg, msgTypes.msg_normal, 1 ],
    formatArgs: args => pipe(
        checkText, // 检查文本，不可为空，过滤危险字符等等
        checkTimestamp // 检查时间戳（此处需要外界传入，因为需要根据此定位消息）
    )(args)
};

// ./C2C/check.js
// 定义各个参数校验规则
/**
 * 校验timestamp
 * @param {Object} args 参数
 */
export const checkTimestamp = args => {
    const { timestamp } = args;
    if (!timestamp) {
        throw new Error('Error in sendMsgs: timestamp is request!');
    }
    return args;
};

# API

IM模块作为整个SDK通信的核心部分，其API还是比较繁多的

# 登录相关

# login

  • @desc - IM登录
  • @param config:Object - 登录配置信息
    • listeners:Object - 事件监听器，用于注册IM消息的回调，详细请移步腾讯IM官方文档
    • options:Object - 环境配置信息
  • @return - Promise<Object>

# isLogin

  • @desc - 判断当前客户端是否是登录状态
  • @return - Boolean

# c2c消息发送相关

# sendMsgsFeedback$:Observable

  • @desc - C2C消息发送状态流，可用于注册消息发送之后的回调

# sendText

  • @desc - 发送文本
  • @param args:Object - 消息参数
    • content:String - 文本
    • timestamp:Number - 时间戳
  • @return - undefined

# sendImg

  • @desc - 发送图片
  • @param args:Object - 消息参数
    • content?:String - 图片消息文本（可选）
    • timestamp:Number - 时间戳
    • imgData:Object - 图片地址等信息
  • @return - undefined

# sendGift

  • @desc - 发送礼物
  • @param args:Object - 消息参数
    • content?:String - 礼物消息文本（可选）
    • timestamp:Number - 时间戳
    • giftData:Object - 礼物id等信息
  • @return - undefined

# sendXHExpression

  • @desc - 发送表情
  • @param args:Object - 消息参数
    • content?:String - 表情消息文本（可选）
    • timestamp:Number - 时间戳
    • giftData:Object - 表情id等信息
  • @return - undefined

# sendLike

  • @desc - 发送点赞消息
  • @param args:Object - 消息参数
    • content?:String - 点赞消息文本（可选）
    • timestamp:Number - 时间戳
  • @return - Promise<Object>

# sendAssess

  • @desc - 发送评价消息
  • @param args:Object - 消息参数
    • content?:String - 评价消息文本（可选）
    • timestamp:Number - 时间戳
    • assessData:Object - 评价信息
  • @return - Promise<Object>

# sendDirective

  • @desc - 发送特定类型的指令消息
  • @param args:Object - 消息参数
    • content?:String - 消息文本（可选）
    • timestamp:Number - 时间戳
    • msgType:String - 消息类型编码
  • @return - Promise<Object>

# 系统消息推送相关

# systemMsgCtor

  • @desc - 系统消息推送消息体构造器
  • @param code:Number - 系统消息编码
  • @param target:Object - 推送目标的信息，包含csId、csNick等
  • @param ext:Object - 额外的信息，将会被包装在 msgBody.data 上，会覆盖
  • @return - Object

# pushMsgTo

  • @desc - 构造系统消息推送至指定用户
  • @param code:Number - 系统消息编码
  • @param target:Object - 推送目标的信息，包含csId、csNick等
  • @param ext:Object - 额外的信息，将会被包装在 msgBody.data 上，会覆盖
  • @return - Promise<Object>

# pushMsg

  • @desc - 推送系统消息至当前sdk内部注册的坐席对象
  • @param code:Number - 系统消息编码
  • @param ext:Object - 额外的信息，将会被包装在 msgBody.data 上，会覆盖
  • @return - Promise<Object>

# 消息接收相关

# parseMsgs

  • @desc - 解析IM消息体
  • @param IMMsgs:MsgElm - 腾讯IM，onMsgNotify接口回调中拿到的消息体
  • @return - Observable<Object>

# getMsgStream$

  • @desc - 获取消息监听的可观测对象
  • @param event:'C2C' | 'system' | 'directive' - 消息类型
  • @return - Subject

# onMsg

  • @desc - 添加消息监听
  • @param event:'C2C' | 'system' | 'directive' - 消息类型
  • @param next:Function - 消息成功接收时的回调
  • @param error?:Function - 消息接收失败时的回调（可选，此处一般是消息解析过程中的错误，可在开发阶段重点关注解析过程）
  • @return - Subscription 可用于注销消息监听

# 其他

# getUserInfo

  • @desc - 获取初始化SDK时注册的用户信息
  • @param - 无
  • @return - Object

# uploadImgs

  • @desc - 上传照片至腾讯服务器
  • @param option:Object - 图片相关数据
    • files:Object - 图片本机路径和大小
    • calcFileInfo:Function - 用于解析图片成base64和buffer（小程序API）
  • @param to:String - 图片上传之后的接收者
  • @return - BehaviorSubject
